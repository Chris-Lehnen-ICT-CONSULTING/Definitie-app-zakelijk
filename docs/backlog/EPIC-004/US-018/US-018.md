def create_tabs():
    tab1, tab2, tab3 = st.tabs([
        "Definitie Generatie",
        "Validatie",
        "Export"
    ])

    with tab1:
        render_definition_tab()
    with tab2:
        render_validation_tab()

        render_export_tab()
