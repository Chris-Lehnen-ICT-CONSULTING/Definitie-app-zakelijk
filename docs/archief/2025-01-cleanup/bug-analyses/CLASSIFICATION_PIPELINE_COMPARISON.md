def classify_begrip(begrip, definitie, context):
    return classifier.classify(begrip, definitie, context)

def generate_definition(request):
    if not request.ontologische_categorie:
        raise ValueError("Category required - call classify_begrip() first")
    return orchestrator.create(request)

    category = classify_begrip(begrip, "", context)
    request = build_request(begrip, category.level)
